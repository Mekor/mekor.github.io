Swift - KVC 和 运行时获取属性列表

2016/11/13

Swift中如何使用KVC

import UIKit

class Person: NSObject {
    var name:String?
    var age: Int = 0
    var title:String?
    
    init(dict:[String:Any]) {
        super.init()
        setValuesForKeys(dict)
    }
    
    
    override func setValue(_ value: Any?, forUndefinedKey key: String) {
        
    }
}

注意:基本数据类型一开始要设置默认值.

Swift中运行时获取属性列表

class Person: NSObject {
    var name:String?
    var age: Int = 0
    var title:String?
    
    init(dict:[String:Any]) {
        super.init()
        setValuesForKeys(dict)
    }
    
    
    override func setValue(_ value: Any?, forUndefinedKey key: String) {
        
    }
    
    class func propertyList() -> [String] {
        var count:UInt32 = 0
        
        let list = class_copyPropertyList(self,&count)
        
        var array:[String] = [];
        
        for i in 0..<Int(count) {
            
            // 获取属性
            let pty = list?[i]
            
            // 获取属性名称
            let cName = property_getName(pty)
            let name = String(utf8String: cName!)
            array.append(name!)
        }
        free(list)
        return array
    }
}

上面是OC写法,可以改进的更Swift一点:

    class func propertyList() -> [String] {
        var count:UInt32 = 0
        
        let list = class_copyPropertyList(self,&count)
        
        var array:[String] = [];
        
        for i in 0..<Int(count) {
            
            guard let pty = list?[i],
                let cName = property_getName(pty),
                let name = String(utf8String: cName)
                else {
                    // 继续遍历下一个
                    continue
            }
            array.append(name)
        }
        free(list)
        return array
    }

通过运行时可以发现:

  1. 基本数据类型如果设置成可选类型则运行时获取不到(在OC中没有可选).... 这就说明了上面的注意点,否则使用KVC会崩溃
  2. 如果属性为private同样获取不到属性(可以获取到ivar),同样会让KVC崩溃

Swift - 函数

2016/11/13

函数的几种情况

// 正常
func sum(x:Int, y:Int) -> Int {
    return x+y
}
print(sum(x: 10, y: 20))

// 去掉形参
func sum(_ x:Int,_ y:Int) -> Int {
    return x+y
}
print(sum(10, 20))

// 外部参数
func sum(num1 x:Int,num2 y:Int) -> Int {
    return x+y
}
print(sum(num1: 10, num2: 20))

// 函数带默认值
func sum(x:Int = 1, y:Int = 2) -> Int {
    return x+y
}
print(sum())

闭包

闭包类似于OC中的block,但应用更广. 与OC对比, OC中block是特殊的函数,在Swift中函数是特殊的闭包

// 最简单的闭包
// 定义
let foo = {
    print("最简单的闭包")
}
// 调用
foo()

// 带参数无返回值
let foo = { (x:Int) -> () in
    print(x)
}
foo(100)

// 带参数有返回值
let foo = { (x:Int) -> Int in
    return x+1
}
foo(10)

block常用在异步执行,回调...Swift中异步跟OC稍微有些区别.

模拟异步加载

    /// 加载数据
    func loadData(completion:@escaping (_ result: [String])->()) {
        
        DispatchQueue.global().async {
            print("耗时操作\(Thread.current)")
            
            DispatchQueue.main.async {
                print("主线程更新\(Thread.current)")
                let json = ["头条","八卦","出大事了"]
                completion(json)
            }
        }
    }

注:关于@escaping参见 Swifter 中@escaping

尾随闭包

        // 如果函数的最后一个参数是闭包,函数参数可以提前结束,最后一个参数直接使用{}包装闭包的代码
        loadData { (JSON) in
            print(JSON)
        }

Swift - 字典中的增删改,遍历与合并

2016/11/12

增删改

var dict:[String:Any] = ["name" : "小明", "age" : 18, "sex":"man"]

// 新增
dict["title"] = "老大"
print(dict)

// 修改
dict["name"] = "赵四"
print(dict)

// 删除
dict.removeValue(forKey: "age")
print(dict)

遍历,合并

// 遍历
for e in dict {
    print(e)
}

for (k,v) in dict {
    print("key:\(k)  value:\(v)")
}
// 合并
var dict1:[String:Any] = ["name" : "李明", "age" : 18, "sex":"m"]
var dict2:[String:Any] = ["name" : "韩梅梅", "age" : 16, "sex":"w"]
// 如果把 dict2 合并到 dict1
for e in dict2 {
    dict1[e.key] = e.value   //dict2[e.key]
}
print("----------\n\(dict1)")

Swift - 数组中的增删改与合并

2016/11/12

第一篇写了数组遍历的几种情况,这篇主要就是数组中的增删改,也是开发中常用的

OC中数组分为可变(NSMutableArray)和不可变(NSArray) 相应的在Swift中就是 var & let

追加元素

首先定义一个可变数组

var array = ["张三","李四","王五"]

追加一个元素

// 追加元素
array.append("赵六")
print(array)

修改元素

通过下标进行修改

array[0] = "赵四"
print(array)

结果:

["赵四", "李四", "王五", "赵六"]

删除

// 删除元素
array.remove(at: 0)
print(array)

结果:

["赵四", "李四", "王五", "赵六"]
["李四", "王五", "赵六"]

remove 的方法由很多,移除首/尾,移除范围...

array.removeAll(keepingCapacity: true)
print("数组中的元素\(array) 数组容量\(array.capacity)")

结果:

数组中的元素[] 数组容量6

数组合并

var arr0 = [1,2,3]
let arr1 = [3,4,5]
arr0 += arr1
print(arr0)

结果:

[1, 2, 3, 3, 4, 5]

注意: 要合并的数组的类型必须一致,如果不一致会出错. 比如:

var arr0:[Any] = [1,2,3]
let arr1:[Any] = ["2",3,4,5]
arr0 += arr1
print(arr0)

Swift - 数组遍历

2016/11/12

最近没怎么更新,主要是最近刚入职新公司.正在熟悉环境中... 好了题目是Swift - 数组遍历 主要是项目要使用 Swift 开发,所以需要熟悉下语法.

传统的下标遍历方法

let array = ["张三","李四","王五"]

// 第一种 下标遍历
for i in 0..<array.count {
    print(array[i])
}

for in 遍历元素

// 第二种 for in 遍历元素
for s in array {
    print(s)
}

enum block遍历

// 第三种 enum block 遍历
for e in array.enumerated() {
    print(e)
}

结果:

(0, "张三")
(1, "李四")
(2, "王五")

我们可以看下e的类型--元组

元组方法遍历下标和内容

由上面第三种方法我们可以知道 e 的类型是元组,那么同样我们可以直接使用元组

// 第四种 元组使用
for (n,s) in array.enumerated() {
    print("\(n) \(s)")
}

结果:

0 张三
1 李四
2 王五

反序遍历

// 第五种 反序遍历
for i in array.reversed() {
    print(i)
}

结果

王五
李四
张三

反序遍历 带索引

// 第六种 反序遍历 带索引
for (n,s) in array.enumerated().reversed() {
    print("\(n) \(s)")
}

结果:

2 王五
1 李四
0 张三

注意: 先枚举再反序

总结:Swift 太灵活了...
